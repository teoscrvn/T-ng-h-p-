-- Load Fluent
local Fluent = loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua"))()
repeat task.wait() until game:IsLoaded()

-- Tạo cửa sổ chính
local Window = Fluent:CreateWindow({
    Title = "Téo 🍎v2 Update",
    SubTitle = "Update Manager",
    TabWidth = 160,
    Size = UDim2.fromOffset(460, 300),
    Acrylic = true,
    Theme = "Amethyst",
    MinimizeKey = Enum.KeyCode.End
})

local Tabs = {
    Main = Window:AddTab({ Title = "Thông báo" }),
    Update = Window:AddTab({ Title = "Update" })
}

-- Tab 1: Thông báo
Tabs.Main:AddLabel("📣 Script này đã update, bạn muốn update không?", { Size = 20 })

Tabs.Main:AddButton({
    Title = "Update",
    Description = "Nhấn để mở trang update",
    Callback = function()
        Window:SelectTab(Tabs.Update) -- Chuyển sang tab Update
    end
})

-- Tab 2: Update
Tabs.Update:AddLabel("🔁 Copy link này dán vào Google để lấy bản update", { Size = 18 })

Tabs.Update:AddButton({
    Title = "📋 Copy link",
    Description = "Copy link update vào clipboard",
    Callback = function()
        local link = ""
        setclipboard(link)
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "✅ Copy thành công",
            Text = "Đã copy link update vào clipboard",
            Duration = 4
        })
    end
})
