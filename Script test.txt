--// Tab 1
local Tab1o = MakeTab({Name = "99 đêm Fake"})

-- Biến điều khiển
local KillAuraEnabled = false
local Range = 20
local TargetMobs = {}

-- Toggle Kill Aura
AddToggle(Tab1o, {
    Name = "🗡 Bật Kill Aura",
    Default = false,
    Callback = function(state)
        KillAuraEnabled = state
        print("Kill Aura: ", state and "BẬT" or "TẮT")
    end
})

-- Slider chỉnh phạm vi
AddSlider(Tab1o, {
    Name = "📏 Phạm vi",
    Min = 5,
    Max = 100,
    Default = 20,
    Callback = function(val)
        Range = val
        print("Phạm vi Kill Aura: " .. val)
    end
})

-- Dropdown chọn sinh vật
AddDropdown(Tab1o, {
    Name = "🎯 Chọn sinh vật",
    Options = {"Bunny", "Wolf", "Alpha Wolf", "Bear"},
    Callback = function(opt)
        TargetMobs = {opt}
        print("Đang chọn: " .. opt)
    end
})

-- Kill Aura loop
task.spawn(function()
    while task.wait(0.2) do
        if KillAuraEnabled and #TargetMobs > 0 then
            local char = game.Players.LocalPlayer.Character
            if char and char:FindFirstChild("HumanoidRootPart") then
                local hrp = char.HumanoidRootPart
                local tool = char:FindFirstChildOfClass("Tool")
                if tool then -- phải cầm vũ khí
                    for _, mob in pairs(workspace:GetDescendants()) do
                        if mob:IsA("Model") and table.find(TargetMobs, mob.Name) then
                            local mobhrp = mob:FindFirstChild("HumanoidRootPart")
                            local hum = mob:FindFirstChild("Humanoid")
                            if mobhrp and hum and hum.Health > 0 then
                                local dist = (mobhrp.Position - hrp.Position).Magnitude
                                if dist <= Range then
                                    -- 👇 Thay event tấn công cho đúng game
                                    if game.ReplicatedStorage:FindFirstChild("Events") 
                                    and game.ReplicatedStorage.Events:FindFirstChild("Attack") then
                                        game.ReplicatedStorage.Events.Attack:FireServer(hum)
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end
    end
end)
