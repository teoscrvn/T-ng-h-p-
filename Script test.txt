------ Tab 1
local Tab1o = MakeTab({Name = "99 đêm Fake"})

-- Kill Aura toggle
local KillAuraEnabled = false
local Range = 15 -- phạm vi tấn công

AddButton(Tab1o, {
    Name = "🗡 Kill Aura",
    Callback = function()
        KillAuraEnabled = not KillAuraEnabled
        if KillAuraEnabled then
            print("Kill Aura BẬT!")
        else
            print("Kill Aura TẮT!")
        end
    end
})

-- Auto Farm Gỗ toggle
local AutoFarmEnabled = false
AddButton(Tab1o, {
    Name = "🌲 Auto Farm Gỗ",
    Callback = function()
        AutoFarmEnabled = not AutoFarmEnabled
        if AutoFarmEnabled then
            print("Auto Farm BẬT!")
        else
            print("Auto Farm TẮT!")
        end
    end
})

-- Loop Kill Aura & Auto Farm
spawn(function()
    while true do
        -- Kill Aura
        if KillAuraEnabled then
            local npcs = workspace:FindFirstChild("NPCs") and workspace.NPCs:GetChildren() or {}
            for _, npc in pairs(npcs) do
                if npc:FindFirstChild("Humanoid") and npc:FindFirstChild("HumanoidRootPart") then
                    local dist = (npc.HumanoidRootPart.Position - game.Players.LocalPlayer.Character.HumanoidRootPart.Position).Magnitude
                    if dist <= Range then
                        if game:GetService("ReplicatedStorage"):FindFirstChild("Events") and game.ReplicatedStorage.Events:FindFirstChild("Attack") then
                            game.ReplicatedStorage.Events.Attack:FireServer(npc.Humanoid)
                        end
                    end
                end
            end
        end
        -- Auto Farm Gỗ
        if AutoFarmEnabled then
            -- Ví dụ: giả lập farm gỗ (cần chỉnh theo game thật)
            print("Đang farm gỗ...")
        end
        wait(0.1)
    end
end)
