------ Tab Kill Aura
local TabKA = MakeTab({Name = "🔥 Kill Aura"})

-- Biến chính
local KillAuraEnabled = false
local Range = 15
local TargetMobs = {}

-- Danh sách entity (ví dụ, bạn thêm vào theo game thật)
local MobList = {"Rabbit", "Wolf", "Alpha Wolf", "Bear", "Deer"}

-- Hàm log popup
local function ShowLog(msg, color)
    local ScreenGui = game.CoreGui:FindFirstChild("AuraLogGui") or Instance.new("ScreenGui", game.CoreGui)
    ScreenGui.Name = "AuraLogGui"

    local InfoLabel = Instance.new("TextLabel")
    InfoLabel.Size = UDim2.new(0, 400, 0, 30)
    InfoLabel.Position = UDim2.new(0.5, -200, 0.85, 0)
    InfoLabel.BackgroundTransparency = 0.3
    InfoLabel.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    InfoLabel.TextColor3 = color or Color3.fromRGB(0, 255, 0)
    InfoLabel.TextScaled = true
    InfoLabel.Text = msg
    InfoLabel.Parent = ScreenGui

    game:GetService("TweenService"):Create(InfoLabel, TweenInfo.new(0.5), {TextTransparency = 1, BackgroundTransparency = 1}):Play()
    game:GetService("Debris"):AddItem(InfoLabel, 1)
end

-- Toggle Kill Aura
AddToggle(TabKA, {
    Name = "🗡 Bật Kill Aura",
    Default = false,
    Callback = function(v)
        KillAuraEnabled = v
        if v then
            ShowLog("⚔️ Kill Aura ON", Color3.fromRGB(0,255,0))
        else
            ShowLog("🛑 Kill Aura OFF", Color3.fromRGB(200,200,200))
        end
    end
})

-- Slider phạm vi
AddSlider(TabKA, {
    Name = "📏 Phạm vi",
    Min = 5,
    Max = 100,
    Default = 15,
    Callback = function(v)
        Range = v
        ShowLog("📐 Phạm vi = "..v, Color3.fromRGB(0,150,255))
    end
})

-- Chọn mob cần kill
for _, mob in ipairs(MobList) do
    AddToggle(TabKA, {
        Name = "🎯 "..mob,
        Default = true,
        Callback = function(v)
            TargetMobs[mob] = v
            ShowLog((v and "✅ Bật " or "❌ Tắt ")..mob, Color3.fromRGB(255,200,0))
        end
    })
end

-- Loop Kill Aura
spawn(function()
    while true do
        if KillAuraEnabled then
            local player = game.Players.LocalPlayer
            local char = player.Character
            if char and char:FindFirstChild("HumanoidRootPart") then
                local hasTool = char:FindFirstChildWhichIsA("Tool")
                if hasTool then
                    local found = false
                    for _, npc in pairs(workspace:GetDescendants()) do
                        if npc:IsA("Model") and TargetMobs[npc.Name] then
                            local hum = npc:FindFirstChild("Humanoid")
                            local root = npc:FindFirstChild("HumanoidRootPart")
                            if hum and root and hum.Health > 0 then
                                local dist = (root.Position - char.HumanoidRootPart.Position).Magnitude
                                if dist <= Range then
                                    found = true
                                    ShowLog("⚔️ Đang đánh "..npc.Name.." ("..math.floor(dist).."m)", Color3.fromRGB(0,255,0))
                                    -- 👇 Đây bạn phải chỉnh theo event attack game thật
                                    game:GetService("ReplicatedStorage"):WaitForChild("Events"):WaitForChild("Attack"):FireServer(hum)
                                end
                            end
                        end
                    end
                    if not found then
                        ShowLog("🔍 Không thấy mục tiêu", Color3.fromRGB(255,255,0))
                    end
                else
                    ShowLog("⚠️ Cần cầm vũ khí", Color3.fromRGB(255,0,0))
                end
            end
        end
        wait(0.5)
    end
end)
